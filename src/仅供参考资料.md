### 一、 完美的“五层情绪引擎”架构方案

为了让眼神“活”起来，必须彻底摒弃“状态A到状态B的简单过渡”，引入时间轴与随机物理扰动：

1. **基础姿态层 (Base Pose):** `expressions.h` 里的静态基础角度（眼皮开合、眉毛基准角度）。
2. **微动作层 (Micro Motion):** 伴随呼吸的低频正弦波形（如眉毛缓慢上下浮动）。
3. **随机扰动层 (Random Event):** 避免死板的机械循环，例如开心的随机眨眼（wink）、恐慌时的随机滴汗、眼球无意识的微小漂移。
4. **二级运动层 (Secondary Overshoot):** 遵循动画十二原则，动作发生时的“过冲与回弹”（例如突然惊醒时眉毛先弹得极高，再回落到正常高度）。
5. **情绪节奏层 (Emotion Rhythm):** 专属的情绪剧情时间轴（State Machine）。例如瞌睡时的“犯困→惊醒→恍惚→再困”的完整周期循环。

### 二、 核心修复与情绪表现力升级教程

#### 1. 修复 Angry 右眉反向与物理颤抖

- **痛点：** 硬件安装时，左右舵机通常是镜像对称的。代码中左眉增加角度是“上扬”，右眉如果也增加相同的角度，视觉上就会变成“下压”。

- 你的项目中已经存在 `EVT_BUTTON_LONG` 触发的 `do_force_return()`。为了保证极高的硬件物理安全和体验闭环，长按“OK”键不应仅仅是重置眼神，而必须确保彻底打断当前所有的动画引擎队列，并**强制回归到主时钟界面**。这一步是确保系统永不锁死的底线。

            s_random_lid_jitter = ((float)(rand() % 50) / 1000.0f) - 0.025f;   /* ±2.5%眼皮微动 */

            float bounce = sin(cfg->brow_anim_phase * 6.0f) * decay * amp;

            off_r = bounce;

            cfg->brow_anim_phase += freq;
            float drift  = sin(cfg->brow_anim_phase * 0.3f) * amp * 0.5f;
            float drift2 = sin(cfg->brow_anim_phase * 0.3f + 1.5f) * amp * 0.5f;
            off_l = -amp + drift;
            off_r = -amp + drift2 * asym;
            break;
        }
        default:
            cfg->brow_anim_phase += freq;
            break;
        }

        /* 二级运动：眉毛余震衰减 */
        static float brow_overdrive_l = 0, brow_overdrive_r = 0;
        brow_overdrive_l *= 0.88f;
        brow_overdrive_r *= 0.88f;
        
        cfg->brow_offset_l = (int8_t)(off_l + brow_overdrive_l);
        cfg->brow_offset_r = (int8_t)(off_r + brow_overdrive_r);
    }

    /* ========== 参数平滑插值 ========== */
    cfg->cur_lid_top    += (cfg->target_lid_top    - cfg->cur_lid_top)    * 0.18f;
    cfg->cur_lid_top_l  += (cfg->target_lid_top_l  - cfg->cur_lid_top_l)  * 0.18f;
    cfg->cur_lid_top_r  += (cfg->target_lid_top_r  - cfg->cur_lid_top_r)  * 0.18f;

    cfg->cur_lid_slope  += (cfg->target_lid_slope  - cfg->cur_lid_slope)  * 0.18f;
    cfg->cur_pupil_scale += (cfg->target_pupil_scale - cfg->cur_pupil_scale) * 0.18f;
    cfg->cur_pupil_type  = cfg->target_pupil_type;

## 一、先对齐：右眉为什么"看似修了还是反"

这是你这个项目最容易踩的坑，先讲透再动手：

**问题根因不在 `expressions.h` 的 `SYM_R()` 数值，而在 `main.cpp` 舵机同步那行的符号。**

- `SYM_L(o) = 90 - o`，`SYM_R(o) = 90 + o`，这只是"静态角度查表"
- 但左右舵机**物理安装是镜像的**（一个朝上装、一个朝下装，或都朝上但连杆反向），所以：
  - 左眉：想"上扬" → `base_l + offset`（offset 正 = 更上）
  - 右眉：想"上扬" → **必须 `base_r - offset`**，否则同向安装下右眉会"下压"

你 v6.1 `main.cpp` loop 里现在是：
```cpp
servo_set_target(
    base_l + g_joy_brow_offset_l + g_eyeCfg.brow_offset_l,
    base_r + g_joy_brow_offset_r + g_eyeCfg.brow_offset_r   // ← 右眉没取反！
);
```

**Angry 倒八字要求：左眉内低外高（内=90-52=38↓，外靠连杆）→ SYM_L(-52)；右眉内低外高（镜像）→ 右舵机外端要"上扬"，对应 SYM_R(52)=142 还是 SYM_R(-52)=38，取决于安装。**

稳妥做法：**`expressions.h` 表里右眉按"视觉意图"写，`main.cpp` 舵机同步那行对右眉取反**。这样无论安装怎么变，改一处就行。

> 📌 结论：`expressions.h` 里 Angry 写成 `SYM_R(52)` 视觉意图是"右眉外端上扬"，但 `main.cpp` 那行必须 `base_r - joy_r - brow_r` 才能真正镜像。你 v9 草稿这点是对的，但 v6.1 源码没改到——这就是"修了还反"的根源。

static void draw_pupil_happy(U8G2* disp, const EyeGeom_t* g) {
    int16_t cx = g->pupil_cx, cy = g->pupil_cy, r = g->pupil_r;
    int16_t hr = r * 3 / 4;
    if (hr < 2) hr = 2;
    /* 上弯弧：饱满笑弧，系数 0.72 更圆 */
    for (int16_t dy = -hr; dy <= 0; dy++) {
        float ratio = 1.0f - (float)(dy + hr) * (dy + hr) / (float)(hr * hr);
        if (ratio < 0) ratio = 0;
        int16_t dx = (int16_t)(r * 0.72f * sqrtf(ratio));  // ← 0.55→0.72
        if (dx < 1) dx = 1;
        disp->drawHLine(cx - dx, cy + dy, dx * 2 + 1);
    }
    /* 下弯弧：笑弧更宽 0.65 */
    for (int16_t dy = 1; dy <= hr; dy++) {
        float ratio = 1.0f - (float)dy * dy / (float)(hr * hr);
        if (ratio < 0) ratio = 0;
        int16_t dx = (int16_t)(r * 0.65f * sqrtf(ratio));  // ← 0.78→0.65 更宽
        if (dx < 1) dx = 1;
        disp->drawHLine(cx - dx, cy + dy, dx * 2 + 1);
    }
    /* 眼底"水光"两点 */
    disp->setDrawColor(1);
    disp->drawDisc(cx - 2, cy + hr/2 - 1, 1, U8G2_DRAW_ALL);
    disp->drawDisc(cx + 2, cy + hr/2 - 1, 1, U8G2_DRAW_ALL);
    disp->setDrawColor(0);
}

### ❤️ Excited — 双相咚-哒 + 爱心 Bezier 化

爱心你 v9 草稿的"宽高分量独立缩放"方向对（咚=宽扁+强，哒=窄高+弱），补两点：

```cpp
static void draw_pupil_heart(U8G2* disp, const EyeGeom_t* g) {
    int16_t cx = g->pupil_cx, cy = g->pupil_cy, r = g->pupil_r;
    if (r < 4) { draw_pupil_normal(disp, g); return; }

    /* 双相心跳 → 宽高分量 */
    uint32_t ms = millis();
    uint32_t bc = ms % 680;
    float bw = 1.0f, bh = 1.0f;
    if (bc < 150) {         /* 咚: 宽扁 */
        float t = (float)bc / 150.0f;
        float s = sinf(t * M_PI);
        bw = 1.0f + s * 0.28f;
        bh = 1.0f - s * 0.18f;
    } else if (bc < 300) {  /* 哒: 窄高 */
        float t = (float)(bc - 150) / 150.0f;
        float s = sinf(t * M_PI);
        bw = 1.0f - s * 0.15f;
        bh = 1.0f + s * 0.25f;
    }

    int16_t r2 = (int16_t)(r / 2 * bw);
    int16_t r3 = (int16_t)(r * 2 / 3 * bh);
    int16_t rf = (int16_t)(r * bh);

    /* 左瓣 / 右瓣 — 圆心更近（更圆润） */
    for (int16_t dy = -r3; dy <= (int16_t)(-r3 * 0.3f); dy++) {
        float ratio = 1.0f - (float)(dy + r3) * (dy + r3) / (float)(r3 * r3 * 0.7f * 0.7f + 1);
        if (ratio < 0) ratio = 0;
        int16_t dx = (int16_t)(r2 * 0.85f * sqrtf(ratio));  // 0.85 让瓣更瘦长
        if (dx > 0) {
            disp->drawHLine(cx - r2 - dx, cy + dy, dx * 2 + 1);
            disp->drawHLine(cx + r2 - dx, cy + dy, dx * 2 + 1);
        }
    }
    /* 中间连接 */
    disp->drawBox(cx - r2, cy - r3/2, r2*2+1, r3/2+2);
    /* 底部尖 — 更早收 */
    for (int16_t dy = 0; dy <= rf; dy++) {
        float t = (float)dy / (float)(rf + 1);
        float w = (int16_t)((float)r2 * 1.1f * (1.0f - t) * (1.0f - t));
        if (w > 0) disp->drawHLine(cx - w, cy + dy, w*2+1);
    }

    /* 高光 */
    disp->setDrawColor(1);
    disp->drawDisc(cx - r2 + 2, cy - r2 + 1, 2, U8G2_DRAW_ALL);
    disp->drawPixel(cx + r2 - 1, cy - r2 + 2);

    /* 环绕星光：咚那帧亮 */
    uint8_t hb = (ms / 340) % 3;
    if (hb == 1) {
        disp->drawPixel(cx + r + 3, cy - r/3);
        disp->drawPixel(cx - r - 3, cy - r/3);
        disp->drawPixel(cx, cy - r - 4);
    }
    disp->setDrawColor(0);
}
```

 *    Layer 4: Emotion Rhythm (情绪节奏: 表情专属剧情)
 *    Layer 5: Servo Behavior (舵机行为: 物理颤抖)
 */

/* ================================================================
 *  缓动工具
 * ================================================================ */
static inline float ease_in(float t)  { return t * t; }
static inline float ease_out(float t) { float i = 1.0f - t; return 1.0f - i * i; }
static inline float ease_in_out(float t) {
    return t < 0.5f ? 2.0f * t * t : 1.0f - powf(-2.0f * t + 2.0f, 2.0f) / 2.0f;
}

static inline int16_t clamp_i16(int16_t v, int16_t lo, int16_t hi) {
    if (v < lo) return lo;
    if (v > hi) return hi;
    return v;
}

/* ---- PUPIL_NORMAL: 标准圆形 ---- */
static void draw_pupil_normal(U8G2* disp, const EyeGeom_t* g) {
    if (g->pupil_r > 0) {
        disp->drawDisc(g->pupil_cx, g->pupil_cy, g->pupil_r, U8G2_DRAW_ALL);
    }
}

/* ---- PUPIL_HEART: 立体爱心 + 环绕星光 (v10 Excited) ---- */
static void draw_pupil_heart(U8G2* disp, const EyeGeom_t* g) {
    int16_t cx = g->pupil_cx, cy = g->pupil_cy, r = g->pupil_r;
    if (r < 4) { draw_pupil_normal(disp, g); return; }
    
    /* v10: 双相心跳脉动 — 宽高独立缩放 */
    uint32_t ms = millis();
    uint32_t beat_cycle = ms % 680;
    float beat_w = 1.0f, beat_h = 1.0f;
    
    if (beat_cycle < 150) {
        /* lub: 强收缩 — 宽扁 */
        float t = (float)beat_cycle / 150.0f;
        float s = sinf(t * M_PI);
        beat_w = 1.0f + s * 0.25f;  /* 宽+25% */
        beat_h = 1.0f - s * 0.15f;  /* 高-15% */
    } else if (beat_cycle < 300) {
        /* dub: 弱回弹 — 窄高 */
        float t = (float)(beat_cycle - 150) / 150.0f;
        float s = sinf(t * M_PI);
        beat_w = 1.0f - s * 0.12f;
        beat_h = 1.0f + s * 0.20f;
    }
    /* 舒张期: 恢复正常 */
    
    int16_t r2 = (int16_t)(r / 2 * beat_w);
    int16_t r3 = (int16_t)(r * 2 / 3 * beat_h);
    int16_t r_full = (int16_t)(r * beat_h);
    
    /* 左右两个上圆弧（心形双瓣） */
    for (int16_t dy = -r3; dy <= 0; dy++) {
        float ratio = 1.0f - (float)(dy + r3) * (dy + r3) / (float)(r3 * r3);
        if (ratio < 0) ratio = 0;
        int16_t dx = (int16_t)(r2 * sqrtf(ratio));
        if (dx > 0) {
            disp->drawHLine(cx - r2 - dx, cy + dy, dx * 2 + 1);
            disp->drawHLine(cx + r2 - dx, cy + dy, dx * 2 + 1);
        }
    }
    
    /* 中间连接填充 */
    disp->drawBox(cx - r2, cy - r3/2, r2 * 2 + 1, r3/2 + 1);
    
    /* 下方尖部：二次曲线圆润收窄 */
    for (int16_t dy = 0; dy <= r_full; dy++) {
        float t = (float)dy / (float)r_full;
        float ratio = (1.0f - t) * (1.0f - t);
        int16_t w = (int16_t)((float)r_full * ratio * beat_w);
        if (w > 0) {
            disp->drawHLine(cx - w, cy + dy, w * 2 + 1);
        }
    }
    
    /* 立体高光 */
    disp->setDrawColor(1);
    int16_t hl_r = (int16_t)(2 * beat_w);
    disp->drawDisc(cx - r2 + 2, cy - r2 + 1, hl_r, U8G2_DRAW_ALL);
    disp->drawPixel(cx + r2 - 1, cy - r2 + 2);
    disp->drawPixel(cx - 1, cy + 2);
    
    /* v10: 环绕星光与心跳同步闪烁 */
    uint8_t beat_phase = (ms / 340) % 3;
    if (beat_phase == 1) {
        int16_t sparkle_r = (int16_t)(r * 1.4f);
        disp->drawPixel(cx + sparkle_r, cy - r/3);
        disp->drawPixel(cx - sparkle_r, cy - r/3);
        disp->drawPixel(cx, cy - r - 3);
        disp->drawPixel(cx + r/2, cy + r/2);
        disp->drawPixel(cx - r/2, cy + r/2);
    }
    disp->setDrawColor(0);
}

/* ---- PUPIL_SLIT: 锐利竖缝 + 怒气红光 (v10 Angry) ---- */
static void draw_pupil_slit(U8G2* disp, const EyeGeom_t* g) {
    int16_t sw = g->pupil_r / 6;
    if (sw < 1) sw = 1;
    
    /* 主竖缝：上下延伸 */
    disp->drawBox(g->pupil_cx - sw, g->pupil_cy - g->pupil_r - 5,
                  sw * 2 + 1, g->pupil_r * 2 + 10);
    
    /* 两侧锐利刀锋边 */
    disp->drawLine(g->pupil_cx - sw - 2, g->pupil_cy - g->pupil_r - 2,
                   g->pupil_cx - sw - 2, g->pupil_cy + g->pupil_r + 2);
    disp->drawLine(g->pupil_cx + sw + 2, g->pupil_cy - g->pupil_r - 2,
                   g->pupil_cx + sw + 2, g->pupil_cy + g->pupil_r + 2);
    
    /* v10: 怒气红光高频闪烁 */
    uint32_t ms = millis();
    uint8_t flash = (ms / 110) % 4;
    if (flash == 0 || flash == 2) {
        disp->setDrawColor(1);
        disp->drawPixel(g->pupil_cx, g->pupil_cy - g->pupil_r - 3);
        disp->drawPixel(g->pupil_cx, g->pupil_cy + g->pupil_r + 3);
        if (g->pupil_r > 5) {
            disp->drawPixel(g->pupil_cx - 1, g->pupil_cy - g->pupil_r - 1);
            disp->drawPixel(g->pupil_cx + 1, g->pupil_cy + g->pupil_r + 1);
            disp->drawPixel(g->pupil_cx - 2, g->pupil_cy + g->pupil_r);
            disp->drawPixel(g->pupil_cx + 2, g->pupil_cy - g->pupil_r);
        }
        disp->setDrawColor(0);
    }
}

/* ---- PUPIL_SHOCK: 十二向电波 + 大小眼配合 (v10 Surprised) ---- */
static void draw_pupil_shock(U8G2* disp, const EyeGeom_t* g) {
    int16_t cx = g->pupil_cx, cy = g->pupil_cy, r = g->pupil_r;
    
    /* 中空圆环瞳孔 */
    int16_t ring_r = r > 2 ? r : 2;
    disp->drawCircle(cx, cy, ring_r, U8G2_DRAW_ALL);
    if (r > 4) {
        disp->setDrawColor(1);
        disp->drawCircle(cx, cy, r - 3, U8G2_DRAW_ALL);
        disp->setDrawColor(0);
    }
    
    /* v10: 十二向电波高速交替闪烁 (28ms切换) */
    uint32_t phase = (millis() / 28) % 4;
    disp->setDrawColor(0);
    
    if (phase == 0) {
        /* 四正方向长电波 */
        disp->drawLine(cx, cy - ring_r - 2, cx, cy - ring_r - 16);
        disp->drawLine(cx, cy + ring_r + 2, cx, cy + ring_r + 16);
        disp->drawLine(cx - ring_r - 2, cy, cx - ring_r - 15, cy);
        disp->drawLine(cx + ring_r + 2, cy, cx + ring_r + 15, cy);
    } else if (phase == 1) {
        /* 四对角方向电波 */
        int16_t d = (int16_t)(ring_r * 0.7f);
        disp->drawLine(cx - ring_r - 1, cy - ring_r - 1, cx - ring_r - 12, cy - ring_r - 12);
        disp->drawLine(cx + ring_r + 1, cy - ring_r - 1, cx + ring_r + 12, cy - ring_r - 12);
        disp->drawLine(cx - ring_r - 1, cy + ring_r + 1, cx - ring_r - 12, cy + ring_r + 12);
        disp->drawLine(cx + ring_r + 1, cy + ring_r + 1, cx + ring_r + 12, cy + ring_r + 12);
    } else if (phase == 2) {
        /* 八向短电波爆发 */
        disp->drawLine(cx - 5, cy - ring_r - 1, cx - 10, cy - ring_r - 10);
        disp->drawLine(cx + 5, cy - ring_r - 1, cx + 10, cy - ring_r - 10);
        disp->drawLine(cx - 5, cy + ring_r + 1, cx - 10, cy + ring_r + 10);
        disp->drawLine(cx + 5, cy + ring_r + 1, cx + 10, cy + ring_r + 10);
        disp->drawLine(cx - ring_r - 1, cy - 5, cx - ring_r - 10, cy - 10);
        disp->drawLine(cx + ring_r + 1, cy - 5, cx + ring_r + 10, cy - 10);
        disp->drawLine(cx - ring_r - 1, cy + 5, cx - ring_r - 10, cy + 10);
        disp->drawLine(cx + ring_r + 1, cy + 5, cx + ring_r + 10, cy + 10);
    } else {
        /* 全向闪烁点（瞳孔地震峰值） */
        disp->drawDisc(cx, cy - ring_r - 14, 1, U8G2_DRAW_ALL);
        disp->drawDisc(cx, cy + ring_r + 14, 1, U8G2_DRAW_ALL);
        disp->drawDisc(cx - ring_r - 14, cy, 1, U8G2_DRAW_ALL);
        disp->drawDisc(cx + ring_r + 14, cy, 1, U8G2_DRAW_ALL);
    }
}

/* ---- PUPIL_HAPPY: 弯月笑眼 + 星光 + wink准备 (v10 Happy) ---- */
static void draw_pupil_happy(U8G2* disp, const EyeGeom_t* g) {
    int16_t cx = g->pupil_cx, cy = g->pupil_cy, r = g->pupil_r;
    int16_t hr = r * 3 / 4;
    if (hr < 2) hr = 2;
    
    /* 上弯弧：饱满笑眼上缘 */
    for (int16_t dy = -hr; dy <= 0; dy++) {
        float ratio = 1.0f - (float)(dy + hr) * (dy + hr) / (float)(hr * hr);
        if (ratio < 0) ratio = 0;
        int16_t dx = (int16_t)(r * 0.78f * sqrtf(ratio));
        if (dx < 1) dx = 1;
        disp->drawHLine(cx - dx, cy + dy, dx * 2 + 1);
    }
    /* 下弯弧：更宽的笑弧 */
    for (int16_t dy = 1; dy <= hr; dy++) {
        float ratio = 1.0f - (float)dy * dy / (float)(hr * hr);
        if (ratio < 0) ratio = 0;
        int16_t dx = (int16_t)(r * 0.78f * sqrtf(ratio));
        if (dx < 1) dx = 1;
        disp->drawHLine(cx - dx, cy + dy, dx * 2 + 1);
    }
    
    /* v10: 多层星光随机闪烁 */
    uint32_t ms = millis();
    uint8_t rand_phase = (ms / 200) % 5;
    disp->setDrawColor(1);
    
    /* 常驻眼角小星光 */
    if (rand_phase != 4) {
        disp->drawDisc(cx - r + 1, cy - r/2, 1, U8G2_DRAW_ALL);
        disp->drawDisc(cx + r - 1, cy - r/2, 1, U8G2_DRAW_ALL);
    }
    /* 随机爆发大星光 */
    if (rand_phase == 1 || rand_phase == 3) {
        disp->drawPixel(cx - r - 2, cy - r/2 - 2);
        disp->drawPixel(cx + r + 2, cy - r/2 - 2);
        disp->drawPixel(cx - r - 1, cy - r/2 + 1);
        disp->drawPixel(cx + r + 1, cy - r/2 + 1);
    }
    /* 眼底亮片 */
    if (rand_phase % 2 == 0) {
        disp->drawPixel(cx - 2, cy + hr/2);
        disp->drawPixel(cx + 2, cy + hr/2);
    }
    disp->setDrawColor(0);
}

/* ---- PUPIL_STAR: 四角星星 sparkle ---- */
static void draw_pupil_star(U8G2* disp, const EyeGeom_t* g) {
    int16_t cx = g->pupil_cx, cy = g->pupil_cy, r = g->pupil_r;
    if (r < 3) { draw_pupil_normal(disp, g); return; }
    
    int16_t cr = r / 3;
    if (cr < 2) cr = 2;
    disp->drawDisc(cx, cy, cr, U8G2_DRAW_ALL);
    
    int16_t spoke_len = r + 2;
    disp->drawLine(cx, cy - cr, cx, cy - spoke_len);
    disp->drawLine(cx, cy + cr, cx, cy + spoke_len);
    disp->drawLine(cx - cr, cy, cx - spoke_len, cy);
    disp->drawLine(cx + cr, cy, cx + spoke_len, cy);
    
    if ((millis() / 60) % 2 == 0) {
        int16_t diag = (int16_t)((float)spoke_len * 0.7f);
        disp->drawLine(cx - cr, cy - cr, cx - diag, cy - diag);
        disp->drawLine(cx + cr, cy - cr, cx + diag, cy - diag);
        disp->drawLine(cx - cr, cy + cr, cx - diag, cy + diag);
        disp->drawLine(cx + cr, cy + cr, cx + diag, cy + diag);
    }
}

const EyeStyle_t* eye_style_get(void) { return &g_style; }

/* ================================================================
 *  eye_geom_compute() — 单帧几何计算
 * ================================================================ */
void eye_geom_compute(EyeGeom_t* geom, const EyeConfig_t* cfg,
                      const EyeStyle_t* style, bool is_left) {
    const int16_t hw = style->eye_w / 2;
    const int16_t hh = style->eye_h / 2;

    geom->hw = hw; geom->hh = hh; geom->is_left = is_left;
    geom->eye_l = cfg->cx - hw; geom->eye_t = cfg->cy - hh;
    geom->eye_r = cfg->cx + hw; geom->eye_b = cfg->cy + hh;

    geom->pupil_r = (int16_t)((float)style->pupil_r * cfg->cur_pupil_scale);
    if (geom->pupil_r < 1) geom->pupil_r = 1;

    int16_t ppx = (int16_t)(cfg->cur_look_x * (float)style->look_max / 127.0f);
    int16_t ppy = (int16_t)(cfg->cur_look_y * (float)style->look_max / 127.0f);
    geom->pupil_cx = clamp_i16(cfg->cx + ppx,
                                geom->eye_l + geom->pupil_r + 1,
                                geom->eye_r - geom->pupil_r - 1);
    geom->pupil_cy = clamp_i16(cfg->cy + ppy,
                                geom->eye_t + geom->pupil_r + 1,
                                geom->eye_b - geom->pupil_r - 1);
    geom->pupil_type = cfg->cur_pupil_type;

    float sp = style->shine_parallax;
    int16_t sx = (int16_t)(cfg->cur_look_x * sp * (float)style->look_max / 127.0f);
    int16_t sy = (int16_t)(cfg->cur_look_y * sp * (float)style->look_max / 127.0f);

    if (style->s1_r > 0) {
        geom->s1_x = clamp_i16(geom->pupil_cx + style->s1_dx + sx,
                                geom->eye_l + style->s1_r + 1,
                                geom->eye_r - style->s1_r - 1);
        geom->s1_y = clamp_i16(geom->pupil_cy + style->s1_dy + sy,
                                geom->eye_t + style->s1_r + 1,
                                geom->eye_b - style->s1_r - 1);
    } else { geom->s1_x = 0; geom->s1_y = 0; }
    if (style->s2_r > 0) {
        geom->s2_x = clamp_i16(geom->pupil_cx + style->s2_dx + sx,
                                geom->eye_l + style->s2_r + 1,
                                geom->eye_r - style->s2_r - 1);
        geom->s2_y = clamp_i16(geom->pupil_cy + style->s2_dy + sy,
                                geom->eye_t + style->s2_r + 1,
                                geom->eye_b - style->s2_r - 1);
    } else { geom->s2_x = 0; geom->s2_y = 0; }
    if (style->s3_r > 0) {
        geom->s3_x = clamp_i16(geom->pupil_cx + style->s3_dx + sx,
                                geom->eye_l + style->s3_r + 1,
                                geom->eye_r - style->s3_r - 1);
        geom->s3_y = clamp_i16(geom->pupil_cy + style->s3_dy + sy,
                                geom->eye_t + style->s3_r + 1,
                                geom->eye_b - style->s3_r - 1);
    } else { geom->s3_x = 0; geom->s3_y = 0; }

    float lid_top, lid_bottom;
    if (cfg->lid > 0.001f) {
        lid_top = cfg->lid; lid_bottom = cfg->lid * 0.5f;
    } else {
        lid_top = is_left ? cfg->cur_lid_top_l : cfg->cur_lid_top_r;
        lid_bottom = cfg->cur_lid_bottom;
    }

    geom->lid_top_base_y = geom->eye_t + (int16_t)((float)(hh * 2 + 4) * lid_top);
    geom->lid_slope_px = (int16_t)(cfg->cur_lid_slope * (float)hh);
    int16_t y_inner = geom->lid_top_base_y + geom->lid_slope_px;
    int16_t y_outer = geom->lid_top_base_y - geom->lid_slope_px;
    geom->lid_y_left = is_left ? y_outer : y_inner;
    geom->lid_y_right = is_left ? y_inner : y_outer;

    if (lid_bottom > 0.001f) {
        geom->lid_bottom_h = (int16_t)((float)(hh + 2) * lid_bottom);
    } else {
        geom->lid_bottom_h = 0;
    }
}

/* ================================================================
 *  绘制函数
 * ================================================================ */
void eye_draw_body(U8G2* disp, const EyeGeom_t* geom) {
    const EyeStyle_t* s = eye_style_get();
    disp->drawRBox(geom->eye_l, geom->eye_t,
                   geom->eye_r - geom->eye_l + 1,
                   geom->eye_b - geom->eye_t + 1, s->eye_radius);
}

void eye_draw_pupil(U8G2* disp, const EyeGeom_t* geom) {
    disp->setDrawColor(0);
    if (geom->pupil_type < PUPIL_COUNT && s_pupil_draw[geom->pupil_type]) {
        s_pupil_draw[geom->pupil_type](disp, geom);
    }
}

void eye_draw_shine(U8G2* disp, const EyeGeom_t* geom) {
    const EyeStyle_t* s = eye_style_get();
    if (geom->pupil_type == PUPIL_HEART ||
        geom->pupil_type == PUPIL_SHOCK ||
        geom->pupil_type == PUPIL_HAPPY) return;

    disp->setDrawColor(1);
    if (s->s1_r > 0) disp->drawDisc(geom->s1_x, geom->s1_y, s->s1_r, U8G2_DRAW_ALL);
    if (s->s2_r > 0) disp->drawDisc(geom->s2_x, geom->s2_y, s->s2_r, U8G2_DRAW_ALL);
    if (s->s3_r > 0) disp->drawDisc(geom->s3_x, geom->s3_y, s->s3_r, U8G2_DRAW_ALL);
}

void eye_draw_lid_mask(U8G2* disp, const EyeGeom_t* geom) {
    disp->setDrawColor(0);
    if (geom->lid_y_left != geom->eye_t || geom->lid_y_right != geom->eye_t) {
        int16_t top = geom->eye_t - 20;
        if (top < 0) top = 0;
        disp->drawTriangle(geom->eye_l - 4, top,
                           geom->eye_r + 4, top,
                           geom->eye_l - 4, geom->lid_y_left);
        disp->drawTriangle(geom->eye_r + 4, top,
                           geom->eye_l - 4, geom->lid_y_left,
                           geom->eye_r + 4, geom->lid_y_right);
    }
    if (geom->lid_bottom_h > 0) {
        disp->drawBox(geom->eye_l - 2,
                      geom->eye_b + 2 - geom->lid_bottom_h,
                      geom->eye_r - geom->eye_l + 5,
                      geom->lid_bottom_h + 2);
    }
    disp->setDrawColor(1);
}

/* ================================================================
 *  v10: 冷汗特效 (Panic)
 * ================================================================ */
static void eye_draw_sweat(U8G2* disp, const EyeGeom_t* geom) {
    uint32_t ms = millis();
    uint32_t cycle = ms % 1800;  /* 1.8秒周期 */
   
    /* 冷汗随机出现 */
    if (cycle < 400) {
        int16_t sx = geom->is_left ? (geom->eye_l + 6) : (geom->eye_r - 6);
        int16_t sy = geom->eye_t + 4 + (int16_t)((float)cycle * 0.02f);
        if (sy < geom->eye_b - 4) {
            disp->setDrawColor(0);
            disp->drawDisc(sx, sy, 2, U8G2_DRAW_ALL);
            disp->drawTriangle(sx-2, sy, sx+2, sy, sx, sy-4);
            disp->setDrawColor(1);
        }
    }
}

/* ================================================================
 *  v10: 眼泪特效 (Sad) — 增强版
 * ================================================================ */
static void eye_draw_tears(U8G2* disp, const EyeGeom_t* geom) {
    int16_t pcx = geom->pupil_cx;
    int16_t pcy = geom->pupil_cy;
    int16_t pr  = geom->pupil_r;

    int16_t tear_origin_y = geom->eye_b - geom->lid_bottom_h;
    if (tear_origin_y < pcy + pr) tear_origin_y = pcy + pr;

    disp->setDrawColor(1);
    int16_t water_y = tear_origin_y - 1;
    disp->drawBox(pcx - pr + 2, water_y, pr * 2 - 4, 2);

    if (millis() % 600 > 300) {
        disp->drawBox(pcx - pr/2 - 1, pcy + pr/3, pr + 2, 2);
    }

    disp->setDrawColor(0);
    const uint32_t cycle_ms = 2200;
