1. **长短按失效/自动回弹无法关闭的逻辑冲突**：你在 `main.cpp` 中定义了一个 `g_revert_deadline_ms`，但每次触发 `EVT_EXPR_SET` 时，逻辑没有重置这个值，导致它一直在后台运行。只要时间一到，系统就强制执行自动回弹逻辑，这覆盖了你的长按锁定需求。
2. **眉毛反向**：舵机安装物理方向不同，会导致你预设的 `SYM_L/SYM_R` 的方向判定与实际相反。
3. **表情表现力（Sleepy/Sad/Skeptic/Surprised）**：目前的 lerp 线性插值太机械了，缺少“突变”的视觉反馈（例如困倦时的垂落加速度、震惊时的瞬间弹动）。

为了彻底解决这些问题，我们需要对“事件逻辑”**和**“表情渲染引擎”进行一次手术式的重构。

### 1. 彻底解决长短按锁定 (`main.cpp` 的修改点)

在 `main.cpp` 的 `process_event` 函数中，我们必须强制引入“长按后永久锁定，直到按下 S1”的逻辑。

```
    if (!g_leftDisp.begin()) return false;
    g_leftDisp.setBusClock(400000); g_leftDisp.setPowerSave(0); return true;
}

bool initRightDisplay() {
    g_rightDisp.getU8x8()->gpio_and_delay_cb = esp32_fast_sw_i2c_gpio_cb;
    if (!g_rightDisp.begin()) return false;
    g_rightDisp.setPowerSave(0); return true;
}

static void do_force_return(void) {
    event_bus_flush(); eye_look_reset(&g_eyeCfg);
    servo_set_target(SERVO_CENTER_DEG, SERVO_CENTER_DEG);
    eye_set_expression(&g_eyeCfg, 0);
    g_revert_deadline_ms = 0;
}

static void process_event(const EventMsg_t* msg) {
    switch (msg->type) {
    case EVT_JOYSTICK_MOVE: {
        eye_set_look(&g_eyeCfg, msg->value_x, msg->value_y);
        g_joy_brow_offset = msg->value_y / 8;
        break;
    }
    case EVT_EXPR_SET: {
        uint8_t idx = msg->value_x;
        if (idx < 8) {
            g_revert_deadline_ms = 0; /* 极其关键：切断任何正在发生的倒计时自动恢复！ */
            eye_set_expression(&g_eyeCfg, idx);
        }
        break;
    }
    case EVT_EXPR_RELEASE: {
        uint16_t held_ms = (uint16_t)msg->value_y;
        if (held_ms < ADC_LONG_PRESS_MS) {
            g_revert_deadline_ms = millis() + 1500;
        } else {
            g_revert_deadline_ms = 0; /* 长按锁定 */
        }
        break;
    }
    case EVT_BUTTON_LONG: do_force_return(); break;
    default: break;
    }
}

static void render_frame(void) {
    if (g_leftReady && g_rightReady) {
        g_leftDisp.clearBuffer(); eye_render(&g_leftDisp, &g_eyeCfg, true);
        g_leftDisp.updateDisplayArea(EYE_TILE_X, EYE_TILE_Y, EYE_TILE_W, EYE_TILE_H);
        
        g_rightDisp.clearBuffer(); eye_render(&g_rightDisp, &g_eyeCfg, false);
        g_rightDisp.updateDisplayArea(EYE_TILE_X, EYE_TILE_Y, EYE_TILE_W, EYE_TILE_H);
    }
}

void setup() {
    Serial.begin(115200); delay(500);
    Wire.begin(PIN_I2C_SDA, PIN_I2C_SCL);
    g_leftReady = initLeftDisplay(); g_rightReady = initRightDisplay();
    event_bus_init();
    eye_config_init(&g_eyeCfg, 64, 32); blink_state_init(&g_blinkState);
    servo_task_init(); input_task_init();
    xTaskCreate(input_task_run, "InputTask", 2048, NULL, 2, &g_inputTaskHandle);
    xTaskCreate(servo_task_run, "ServoTask", 1536, NULL, 2, &g_servoTaskHandle);
}

void loop() {
    uint32_t now = millis();
    EventMsg_t msg;
    while (event_bus_pop(&msg, 0)) process_event(&msg);

    if (now - g_last_frame_ms < FRAME_INTERVAL_MS) { vTaskDelay(1); return; }
    g_last_frame_ms = now;

    if (g_revert_deadline_ms > 0 && now >= g_revert_deadline_ms) {
        g_revert_deadline_ms = 0;
        eye_set_expression(&g_eyeCfg, 0); 
    }

    eye_look_update(&g_eyeCfg);
    eye_expr_update(&g_eyeCfg, now);
    blink_state_update(&g_blinkState, &g_eyeCfg, now);

    /* --- 眉毛高级物理震颤引擎 --- */
    int8_t base_l = 90, base_r = 90;
    if (g_eyeCfg.active_expr < 8) {
        base_l = EXPRESSIONS[g_eyeCfg.active_expr].brow_left;
        base_r = EXPRESSIONS[g_eyeCfg.active_expr].brow_right;
    }

    /* Angry: 暴怒高频震颤 */
    if (g_eyeCfg.active_expr == 2) {
        int8_t tremble = random(-4, 5); // 剧烈抖动
        servo_set_target(base_l - g_joy_brow_offset + tremble, base_r + g_joy_brow_offset + tremble);
    } 
    /* Sad: 委屈微弱抽泣波 */
    else if (g_eyeCfg.active_expr == 3) {
        int8_t wave = (int8_t)(sin(now * 0.005f) * 4.0f);
        servo_set_target(base_l - g_joy_brow_offset + wave, base_r + g_joy_brow_offset - wave);
    } 
    /* 正常状态融合摇杆 */
    else {
        servo_set_target(base_l - g_joy_brow_offset, base_r + g_joy_brow_offset);
    }

    render_frame();
}
```

```
#include "eye_renderer.h"
#include "expressions.h"
#include <math.h>

static inline float ease_in(float t)  { return t * t; }
static inline float ease_out(float t) { float i = 1.0f - t; return 1.0f - i * i; }

void eye_config_init(EyeConfig_t* cfg, uint8_t cx, uint8_t cy) {
    cfg->cx = cx; cfg->cy = cy; cfg->lid = 0.0f;
    cfg->target_look_x = 0; cfg->target_look_y = 0; cfg->cur_look_x = 0.0f; cfg->cur_look_y = 0.0f;
    cfg->active_expr = 255;
    cfg->target_lid_top = 0.0f; cfg->target_lid_top_l = 0.0f; cfg->target_lid_top_r = 0.0f;
    cfg->target_lid_bottom = 0.0f; cfg->target_lid_slope = 0.0f;
    cfg->target_pupil_scale = 1.0f; cfg->target_pupil_type = PUPIL_NORMAL; cfg->cur_pupil_type = PUPIL_NORMAL;
    cfg->cur_lid_top = 0.0f; cfg->cur_lid_top_l = 0.0f; cfg->cur_lid_top_r = 0.0f;
    cfg->cur_lid_bottom = 0.0f; cfg->cur_lid_slope = 0.0f; cfg->cur_pupil_scale = 1.0f;
    cfg->anim_peak_scale = 0.0f; cfg->anim_start_ms = 0; cfg->anim_duration_ms = 0;
}

void eye_set_look(EyeConfig_t* cfg, int8_t x, int8_t y) { cfg->target_look_x = x; cfg->target_look_y = y; }
void eye_look_update(EyeConfig_t* cfg) {
    cfg->cur_look_x += ((float)cfg->target_look_x - cfg->cur_look_x) * LOOK_SMOOTH_FACTOR;
    cfg->cur_look_y += ((float)cfg->target_look_y - cfg->cur_look_y) * LOOK_SMOOTH_FACTOR;
}
void eye_look_reset(EyeConfig_t* cfg) { cfg->target_look_x = 0; cfg->target_look_y = 0; }

void eye_set_expression(EyeConfig_t* cfg, uint8_t expr_index) {
    if (expr_index >= 8) return;
    const ExpressionDef_t* expr = &EXPRESSIONS[expr_index];
    cfg->active_expr = expr_index;
    cfg->target_lid_top = expr->lid_top; cfg->target_lid_top_l = expr->lid_top_l; cfg->target_lid_top_r = expr->lid_top_r;
    cfg->target_lid_bottom = expr->lid_bottom; cfg->target_lid_slope = expr->lid_slope;
    cfg->target_pupil_type = expr->pupil_type;

    if (expr->anim_peak > 0.001f || expr->anim_peak < -0.001f) {
        cfg->target_pupil_scale = expr->anim_peak; cfg->anim_peak_scale = expr->anim_peak;
        cfg->anim_start_ms = millis(); cfg->anim_duration_ms = expr->anim_ms;
    } else {
        cfg->target_pupil_scale = expr->pupil_scale; cfg->anim_peak_scale = 0.0f;
    }
}

void eye_expr_update(EyeConfig_t* cfg, uint32_t now_ms) {
    if (cfg->anim_peak_scale > 0.001f || cfg->anim_peak_scale < -0.001f) {
        if (now_ms - cfg->anim_start_ms >= cfg->anim_duration_ms) {
            cfg->anim_peak_scale = 0.0f;
            if (cfg->active_expr < 8) cfg->target_pupil_scale = EXPRESSIONS[cfg->active_expr].pupil_scale;
        }
    }

    float tgt_top_l = cfg->target_lid_top_l;
    float tgt_top_r = cfg->target_lid_top_r;

    /* Sleepy 瞌睡引擎: 打破常规 lerp, 强制直接赋予正弦数学模型 */
    if (cfg->active_expr == 5) {
        float t = fmod(now_ms / 2000.0f, 1.0f); // 2秒一个呼吸周期 0->1
        float drop = ease_in(t); // 缓慢下落
        if (t > 0.85f) drop = 0.0f; // 最后15%突然惊醒
        tgt_top_l = 0.45f + 0.45f * drop; // 从 0.45 掉到 0.9
        tgt_top_r = tgt_top_l;
        /* 直接覆盖防止被lerp稀释 */
        cfg->cur_lid_top_l = tgt_top_l;
        cfg->cur_lid_top_r = tgt_top_r;
    } else {
        cfg->cur_lid_top_l += (tgt_top_l - cfg->cur_lid_top_l) * LOOK_SMOOTH_FACTOR;
        cfg->cur_lid_top_r += (tgt_top_r - cfg->cur_lid_top_r) * LOOK_SMOOTH_FACTOR;
    }

    cfg->cur_lid_top += (cfg->target_lid_top - cfg->cur_lid_top) * LOOK_SMOOTH_FACTOR;
    cfg->cur_lid_bottom += (cfg->target_lid_bottom - cfg->cur_lid_bottom) * LOOK_SMOOTH_FACTOR;
    cfg->cur_lid_slope += (cfg->target_lid_slope - cfg->cur_lid_slope) * LOOK_SMOOTH_FACTOR;
    cfg->cur_pupil_scale += (cfg->target_pupil_scale - cfg->cur_pupil_scale) * LOOK_SMOOTH_FACTOR;
    cfg->cur_pupil_type = cfg->target_pupil_type;
}

void blink_state_init(BlinkState_t* state) {
    state->phase = BLINK_IDLE; state->next_blink_ms = millis() + random(1500, 3001);
}

void blink_state_update(BlinkState_t* state, EyeConfig_t* cfg, uint32_t now_ms) {
    switch (state->phase) {
    case BLINK_IDLE:
        cfg->lid = 0.0f;
        if (now_ms >= state->next_blink_ms) {
            state->phase = BLINK_CLOSING; state->phase_start_ms = now_ms; state->phase_duration_ms = BLINK_CLOSING_MS;
        }
        break;
    case BLINK_CLOSING: {
        float t = (float)(now_ms - state->phase_start_ms) / state->phase_duration_ms; if (t > 1.0f) t = 1.0f;
        cfg->lid = ease_in(t);
        if (t >= 1.0f) { state->phase = BLINK_HOLD; state->phase_start_ms = now_ms; state->phase_duration_ms = BLINK_HOLD_MS; }
        break;
    }
    case BLINK_HOLD:
        cfg->lid = 1.0f;
        if (now_ms - state->phase_start_ms >= state->phase_duration_ms) {
            state->phase = BLINK_OPENING; state->phase_start_ms = now_ms; state->phase_duration_ms = BLINK_OPENING_MS;
        }
        break;
    case BLINK_OPENING: {
        float t = (float)(now_ms - state->phase_start_ms) / state->phase_duration_ms; if (t > 1.0f) t = 1.0f;
        cfg->lid = 1.0f - ease_out(t);
        if (t >= 1.0f) {
            cfg->lid = 0.0f;
            if (random(12) == 0) { state->phase = BLINK_CLOSING; state->phase_start_ms = now_ms; state->phase_duration_ms = BLINK_CLOSING_MS; } 
            else { state->phase = BLINK_IDLE; state->next_blink_ms = now_ms + random(BLINK_INTERVAL_MIN, BLINK_INTERVAL_MAX + 1); }
        }
        break;
    }
    }
}

static inline int16_t clamp_i16(int16_t v, int16_t lo, int16_t hi) { if (v < lo) return lo; if (v > hi) return hi; return v; }

void eye_render(U8G2* disp, EyeConfig_t* cfg, bool is_left) {
    int16_t hw = EYE_W / 2; int16_t hh = EYE_H / 2;
    int16_t eye_l = cfg->cx - hw; int16_t eye_t = cfg->cy - hh;
    int16_t eye_r = cfg->cx + hw; int16_t eye_b = cfg->cy + hh;

    disp->setDrawColor(1);
    disp->drawRBox(eye_l, eye_t, EYE_W, EYE_H, EYE_RADIUS);

    int16_t pupil_r = (int16_t)((float)PUPIL_R * cfg->cur_pupil_scale);
    if (pupil_r < 1) pupil_r = 1;
    int16_t ppx = clamp_i16((int16_t)(cfg->cur_look_x * LOOK_MAX / 127.0f), -(hw - pupil_r - 2), (hw - pupil_r - 2));
    int16_t ppy = clamp_i16((int16_t)(cfg->cur_look_y * LOOK_MAX / 127.0f), -(hh - pupil_r - 2), (hh - pupil_r - 2));
    int16_t pcx = cfg->cx + ppx; int16_t pcy = cfg->cy + ppy;

    disp->setDrawColor(0);

    /* === 多态瞳孔 === */
    if (cfg->cur_pupil_type == PUPIL_HEART) {
        /* 高精度像素爱心 */
        int16_t hr = pupil_r;
        disp->drawDisc(pcx - hr/2, pcy - hr/3, hr/2+1, U8G2_DRAW_ALL);
        disp->drawDisc(pcx + hr/2, pcy - hr/3, hr/2+1, U8G2_DRAW_ALL);
        disp->drawTriangle(pcx - hr, pcy - hr/3 + 2, pcx + hr, pcy - hr/3 + 2, pcx, pcy + hr + 2);
    }
    else if (cfg->cur_pupil_type == PUPIL_HAPPY) {
        /* ^ ^ 欢快眼 */
        disp->drawLine(pcx-pupil_r, pcy+2, pcx-pupil_r/2, pcy-pupil_r/2+2); disp->drawLine(pcx-pupil_r/2, pcy-pupil_r/2+2, pcx, pcy+2);
        disp->drawLine(pcx, pcy+2, pcx+pupil_r/2, pcy-pupil_r/2+2); disp->drawLine(pcx+pupil_r/2, pcy-pupil_r/2+2, pcx+pupil_r, pcy+2);
        disp->drawLine(pcx-pupil_r, pcy+3, pcx-pupil_r/2, pcy-pupil_r/2+3); disp->drawLine(pcx-pupil_r/2, pcy-pupil_r/2+3, pcx, pcy+3);
        disp->drawLine(pcx, pcy+3, pcx+pupil_r/2, pcy-pupil_r/2+3); disp->drawLine(pcx+pupil_r/2, pcy-pupil_r/2+3, pcx+pupil_r, pcy+3);
    }
    else if (cfg->cur_pupil_type == PUPIL_SHOCK) {
        /* 空心震惊瞳孔 + 电波 */
        disp->drawCircle(pcx, pcy, pupil_r/2, U8G2_DRAW_ALL);
        disp->drawDisc(pcx, pcy, 1, U8G2_DRAW_ALL); 
        if ((millis() / 40) % 2 == 0) {
            disp->drawLine(pcx-6, pcy-6, pcx-14, pcy-14); disp->drawLine(pcx+6, pcy-6, pcx+14, pcy-14);
            disp->drawLine(pcx-6, pcy+6, pcx-14, pcy+14); disp->drawLine(pcx+6, pcy+6, pcx+14, pcy+14);
        } else {
            disp->drawLine(pcx-4, pcy-8, pcx-10, pcy-16); disp->drawLine(pcx+4, pcy-8, pcx+10, pcy-16);
            disp->drawLine(pcx-4, pcy+8, pcx-10, pcy+16); disp->drawLine(pcx+4, pcy+8, pcx+10, pcy+16);
        }
    }
    else if (cfg->cur_pupil_type == PUPIL_SLIT) {
        int16_t sw = pupil_r / 3; if (sw < 1) sw = 1;
        disp->drawBox(pcx - sw, pcy - pupil_r, sw * 2, pupil_r * 2);
    }
    else disp->drawDisc(pcx, pcy, pupil_r, U8G2_DRAW_ALL);

    /* === Sad 泪滴引擎 === */
    if (cfg->active_expr == 3) {
        /* 眼泪汪汪底部积水 */
        disp->setDrawColor(1);
        disp->drawBox(pcx - pupil_r + 2, pcy + pupil_r/2, pupil_r*2 - 4, 3);
        
        /* 巨大滑落黑泪 */
        disp->setDrawColor(0);
        uint32_t t_tear = millis() % 1200;
        int16_t tear_y = pcy + pupil_r + (int16_t)(t_tear * 25 / 1200);
        if (tear_y < eye_b) {
            int16_t tear_x = pcx + (is_left ? -5 : 5);
            disp->drawDisc(tear_x, tear_y, 3, U8G2_DRAW_ALL);
            disp->drawTriangle(tear_x - 3, tear_y, tear_x + 3, tear_y, tear_x, tear_y - 6);
        }
    }

    disp->setDrawColor(1);
```

### 一、核心Bug修复：ADC键盘长短按失效 + 自动回弹异常
#### 问题根源
1. **缺失释放事件**：现有ADC键盘仅在按下时发送`EVT_EXPR_SET`，释放时完全静默，导致主循环`EVT_EXPR_RELEASE`分支永不执行，长短按判定完全失效。
2. **回弹计时未重置**：按下新表情时未清空自动回弹计时器，旧计时到期后会强制把新表情切回Normal。
3. **时长传递链路未打通**：事件结构体已支持int16_t时长，但未实际填充和使用。

#### 修复代码
##### 1. 替换 `input_task.cpp` 中ADC键盘检测逻辑
```cpp
/* ---- 5. ADC 键盘检测 (边沿触发, <50ms 响应) ---- */
{
    /* 采集 ADC 键盘 (8次平均去噪) */
    int32_t kb_sum = 0;
    for (uint8_t i = 0; i < 8; i++) {
        kb_sum += analogRead(PIN_ADC_KEYBOARD);
    }
    uint16_t kb_avg = (uint16_t)(kb_sum / 8);
    /* 查表: ADC → 键值 (1-8) 或 0=NONE */
    uint8_t kb_current = adc_lookup_expr(kb_avg);

    /* 边沿去抖: 值变化 → 重置计时器 */
    if (kb_current != adc_current_raw) {
        adc_current_raw = kb_current;
        adc_debounce_ms = millis();
    }

    /* 去抖窗口到期 → 确认边沿 */
    uint32_t kb_now = millis();
    if (kb_now - adc_debounce_ms >= ADC_KEY_DEBOUNCE_MS) {
        if (adc_current_raw != adc_last_stable) {
            /* 按键按下: 推送 EVT_EXPR_SET */
            if (adc_current_raw != ADC_KEY_NONE) {
                EventMsg_t kmsg;
                kmsg.type    = EVT_EXPR_SET;
                kmsg.value_x = adc_current_raw - 1;  /* 转为 0-based 索引 */
                kmsg.value_y = 0;
                event_bus_push(&kmsg);
                
                // 记录按下时刻和按键，用于释放时计算时长
                adc_press_start_ms = kb_now;
                adc_pressed_key = adc_current_raw;

                Serial.print(F("[KEY] S"));
                Serial.print((int)adc_current_raw);
                Serial.print(F(" pressed (ADC="));
                Serial.print(kb_avg);
                Serial.print(F(") → expr="));
                Serial.println(EXPRESSIONS[adc_current_raw - 1].name);
            }
            /* 按键释放: 推送 EVT_EXPR_RELEASE，携带持有时长 */
            else {
                if (adc_pressed_key != ADC_KEY_NONE) {
                    uint16_t held_ms = kb_now - adc_press_start_ms;
                    EventMsg_t kmsg;
                    kmsg.type    = EVT_EXPR_RELEASE;
                    kmsg.value_x = adc_pressed_key - 1;  // 0-based索引
                    kmsg.value_y = (int16_t)held_ms;     // 完整传递毫秒时长
                    event_bus_push(&kmsg);

                    Serial.print(F("[KEY] S"));
                    Serial.print((int)adc_pressed_key);
                    Serial.print(F(" released, held="));
                    Serial.print(held_ms);
                    Serial.println(F("ms"));

                    adc_pressed_key = ADC_KEY_NONE;
                }
            }
            adc_last_stable = adc_current_raw;
        }
    }
}
```

```cpp
void eye_expr_update(EyeConfig_t* cfg, uint32_t now_ms) {
    if (cfg->anim_peak_scale > 0.001f || cfg->anim_peak_scale < -0.001f) {
        uint32_t elapsed = now_ms - cfg->anim_start_ms;
        if (elapsed >= cfg->anim_duration_ms) {
            cfg->anim_peak_scale = 0.0f;
            if (cfg->active_expr < 8) {
                cfg->target_pupil_scale = EXPRESSIONS[cfg->active_expr].pupil_scale;
            }
        }
    }

    // 眉毛微动相位累计
    cfg->brow_anim_phase_ms += FRAME_INTERVAL_MS;
    float phase = cfg->brow_anim_phase_ms * 0.001f;

    switch(cfg->active_expr) {
        case 1: // Happy: 轻快小挑眉
            cfg->brow_offset_l = sin(phase * 6.0f) * 3.0f;
            cfg->brow_offset_r = sin(phase * 6.0f) * 3.0f;
            break;
        case 2: // Angry: 高频怒火颤抖
            cfg->brow_offset_l = sin(phase * 20.0f) * 2.5f;
            cfg->brow_offset_r = sin(phase * 20.0f + 0.3f) * 2.5f;
            break;
        case 3: // Sad: 缓慢抽泣浮动
            cfg->brow_offset_l = sin(phase * 1.5f) * 2.0f;
            cfg->brow_offset_r = sin(phase * 1.5f) * 2.0f;
            break;
        case 4: // Surprised: 弹跳余震
            cfg->brow_offset_l = sin(phase * 8.0f) * 1.5f * exp(-phase * 2);
            cfg->brow_offset_r = sin(phase * 8.0f) * 1.5f * exp(-phase * 2);
            break;
        case 5: // Sleepy: 疲惫下垂微晃
            cfg->brow_offset_l = -1.5f + sin(phase * 0.8f) * 1.0f;
            cfg->brow_offset_r = -1.5f + sin(phase * 0.8f) * 1.0f;
            break;
        case 6: // Skeptic: 右眉轻挑不屑
            cfg->brow_offset_l = 0;
            cfg->brow_offset_r = sin(phase * 2.0f) * 1.5f;
            break;
        case 7: // Excited: 心跳式挑眉
            cfg->brow_offset_l = fabs(sin(phase * 3.0f)) * 4.0f;
            cfg->brow_offset_r = fabs(sin(phase * 3.0f)) * 4.0f;
            break;
        default:
            cfg->brow_offset_l = 0;
            cfg->brow_offset_r = 0;
            break;
    }

    /* Sleepy 锯齿缓动瞌睡引擎 — 缓慢合→惊醒→再合 */
    if (cfg->active_expr == 5) {
        cfg->sleepy_phase_ms += FRAME_INTERVAL_MS;
        uint32_t cycle_ms = 4000;
        uint32_t t = cfg->sleepy_phase_ms % cycle_ms;
        float sleepy_lid;
        
        if (t < 3000) {
            // 3秒缓慢合上80%
            sleepy_lid = (float)t / 3000.0f * 0.8f;
        } else if (t < 3300) {
            // 0.3秒惊醒弹开至20%
            float p = (float)(t - 3000) / 300.0f;
            sleepy_lid = 0.8f - p * 0.6f;
        } else {
            // 0.7秒再次缓慢合上
            float p = (float)(t - 3300) / 700.0f;
            sleepy_lid = 0.2f + p * 0.6f;
        }
        cfg->target_lid_top = sleepy_lid;
        cfg->target_lid_bottom = 0.1f + sleepy_lid * 0.5f;
    }

    cfg->cur_lid_top     += (cfg->target_lid_top     - cfg->cur_lid_top)     * LOOK_SMOOTH_FACTOR;
    cfg->cur_lid_top_l   += (cfg->target_lid_top_l   - cfg->cur_lid_top_l)   * LOOK_SMOOTH_FACTOR;
    cfg->cur_lid_top_r   += (cfg->target_lid_top_r   - cfg->cur_lid_top_r)   * LOOK_SMOOTH_FACTOR;
    cfg->cur_lid_bottom  += (cfg->target_lid_bottom  - cfg->cur_lid_bottom)  * LOOK_SMOOTH_FACTOR;
    cfg->cur_lid_slope   += (cfg->target_lid_slope   - cfg->cur_lid_slope)   * LOOK_SMOOTH_FACTOR;
    cfg->cur_pupil_scale += (cfg->target_pupil_scale - cfg->cur_pupil_scale) * LOOK_SMOOTH_FACTOR;
    cfg->cur_pupil_type = cfg->target_pupil_type;
}
```

# 长短按（必须作为第一优先级修复）

从你目前源码来看，`EventMsg_t` 已经把 `value_x/value_y` 扩展到了 `int16_t`，这一部分已经修复。

因此，请要求 Codex **不要再修改 `event_bus.h`**，而是把重点放在完整的事件流排查上：

- 检查 `input_task.cpp` 是否重复发送 `EVT_EXPR_SET`、`EVT_EXPR_RELEASE`。
- 检查 `main.cpp` 是否在收到新的 `EVT_EXPR_SET` 后没有正确重置 `g_revert_deadline_ms`，导致旧的自动回弹计时器仍然生效。
- 检查 `force_return.cpp` 是否在释放按键时又额外发送了恢复事件。

- 建立完整的状态机（Idle → ShortHold → LongHold → Released），确保短按、长按互斥，不允许多个模块分别维护表情状态。
